# Advanced Patterns

This guide covers advanced usage patterns and techniques for building complex forms with React Any Shape Form.

## Conditional Fields

Show or hide fields based on other field values.

### Basic Conditional Rendering

```tsx
const MyForm = useForm({
  accountType: "personal",
  companyName: "",
  taxId: ""
});

function ConditionalForm() {
  const accountType = MyForm.useWatch("accountType");

  return (
    <MyForm onSubmit={(state) => console.log(state)}>
      <MyForm.Item name="accountType">
        {({ value, onChange }) => (
          <select value={value} onChange={(e) => onChange(e.target.value)}>
            <option value="personal">Personal</option>
            <option value="business">Business</option>
          </select>
        )}
      </MyForm.Item>

      {accountType === "business" && (
        <>
          <MyForm.Item name="companyName">
            {({ value, onChange }) => (
              <input
                placeholder="Company Name"
                value={value}
                onChange={(e) => onChange(e.target.value)}
              />
            )}
          </MyForm.Item>

          <MyForm.Item name="taxId">
            {({ value, onChange }) => (
              <input
                placeholder="Tax ID"
                value={value}
                onChange={(e) => onChange(e.target.value)}
              />
            )}
          </MyForm.Item>
        </>
      )}

      <button type="submit">Submit</button>
    </MyForm>
  );
}
```

## Multi-Step Forms

Break large forms into multiple steps.

### Basic Wizard

```tsx
const MyForm = useForm({
  // Step 1
  firstName: "",
  lastName: "",
  email: "",

  // Step 2
  address: "",
  city: "",
  zipCode: "",

  // Step 3
  cardNumber: "",
  expiryDate: "",
  cvv: ""
});

function MultiStepForm() {
  const [step, setStep] = useState(1);

  const validateStep = async (step: number) => {
    const fieldsToValidate = {
      1: ["firstName", "lastName", "email"],
      2: ["address", "city", "zipCode"],
      3: ["cardNumber", "expiryDate", "cvv"]
    }[step] as Array<keyof typeof MyForm.formApi>;

    try {
      await MyForm.formApi.validateFields(fieldsToValidate);
      return true;
    } catch {
      return false;
    }
  };

  const handleNext = async () => {
    const isValid = await validateStep(step);
    if (isValid) {
      setStep(step + 1);
    }
  };

  const handleBack = () => {
    setStep(step - 1);
  };

  const handleSubmit = async (state: any) => {
    console.log("Final submission:", state);
    // Submit to server
  };

  return (
    <div>
      <div className="steps">
        <div className={step >= 1 ? "active" : ""}>1. Personal Info</div>
        <div className={step >= 2 ? "active" : ""}>2. Address</div>
        <div className={step >= 3 ? "active" : ""}>3. Payment</div>
      </div>

      <MyForm onSubmit={handleSubmit}>
        {step === 1 && (
          <div>
            <h2>Personal Information</h2>
            <MyForm.Item name="firstName" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="First Name" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="lastName" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="Last Name" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="email" rules={[{ required: true }, { type: "email" }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="Email" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
          </div>
        )}

        {step === 2 && (
          <div>
            <h2>Address</h2>
            <MyForm.Item name="address" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="Street Address" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="city" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="City" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="zipCode" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="ZIP Code" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
          </div>
        )}

        {step === 3 && (
          <div>
            <h2>Payment Information</h2>
            <MyForm.Item name="cardNumber" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="Card Number" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="expiryDate" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="MM/YY" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
            <MyForm.Item name="cvv" rules={[{ required: true }]}>
              {({ value, onChange, errors }) => (
                <div>
                  <input placeholder="CVV" value={value} onChange={(e) => onChange(e.target.value)} />
                  {errors[0] && <span>{errors[0].errorText}</span>}
                </div>
              )}
            </MyForm.Item>
          </div>
        )}

        <div className="actions">
          {step > 1 && <button type="button" onClick={handleBack}>Back</button>}
          {step < 3 && <button type="button" onClick={handleNext}>Next</button>}
          {step === 3 && <button type="submit">Submit</button>}
        </div>
      </MyForm>
    </div>
  );
}
```

### Persistent State Across Routes

Use `createForm` to persist form state when navigating between routes:

```tsx
// forms/CheckoutForm.ts
import { createForm } from 'react-any-shape-form';

export const CheckoutForm = createForm({
  // Personal info
  firstName: "",
  lastName: "",
  email: "",

  // Shipping
  address: "",
  city: "",
  zipCode: "",

  // Payment
  cardNumber: "",
  expiryDate: ""
});

// pages/PersonalInfo.tsx
export function PersonalInfoPage() {
  return (
    <CheckoutForm>
      <CheckoutForm.Item name="firstName">
        {({ value, onChange }) => (
          <input value={value} onChange={(e) => onChange(e.target.value)} />
        )}
      </CheckoutForm.Item>
      {/* ... other fields ... */}
      <Link to="/checkout/shipping">Next: Shipping</Link>
    </CheckoutForm>
  );
}

// pages/Shipping.tsx
export function ShippingPage() {
  // Form state persists from PersonalInfoPage
  const firstName = CheckoutForm.useWatch("firstName");

  return (
    <div>
      <p>Shipping for {firstName}</p>
      <CheckoutForm>
        <CheckoutForm.Item name="address">
          {({ value, onChange }) => (
            <input value={value} onChange={(e) => onChange(e.target.value)} />
          )}
        </CheckoutForm.Item>
        {/* ... other fields ... */}
        <Link to="/checkout/payment">Next: Payment</Link>
      </CheckoutForm>
    </div>
  );
}
```

## Dynamic Field Dependencies

Fields that depend on and affect each other.

### Auto-Calculate Field

```tsx
const MyForm = useForm({
  price: 0,
  quantity: 1,
  tax: 0,
  total: 0
});

function InvoiceForm() {
  const price = MyForm.useWatch("price");
  const quantity = MyForm.useWatch("quantity");
  const tax = MyForm.useWatch("tax");

  // Auto-calculate total whenever dependencies change
  React.useEffect(() => {
    const subtotal = price * quantity;
    const total = subtotal + (subtotal * tax / 100);
    MyForm.formApi.setFieldValue("total", total);
  }, [price, quantity, tax]);

  return (
    <MyForm onSubmit={(state) => console.log(state)}>
      <MyForm.Item name="price">
        {({ value, onChange }) => (
          <input
            type="number"
            placeholder="Price"
            value={value}
            onChange={(e) => onChange(Number(e.target.value))}
          />
        )}
      </MyForm.Item>

      <MyForm.Item name="quantity">
        {({ value, onChange }) => (
          <input
            type="number"
            placeholder="Quantity"
            value={value}
            onChange={(e) => onChange(Number(e.target.value))}
          />
        )}
      </MyForm.Item>

      <MyForm.Item name="tax">
        {({ value, onChange }) => (
          <input
            type="number"
            placeholder="Tax %"
            value={value}
            onChange={(e) => onChange(Number(e.target.value))}
          />
        )}
      </MyForm.Item>

      <MyForm.Item name="total">
        {({ value }) => (
          <div>
            <strong>Total: \${value.toFixed(2)}</strong>
          </div>
        )}
      </MyForm.Item>

      <button type="submit">Submit</button>
    </MyForm>
  );
}
```

### Linked Select Dropdowns

Country/State/City cascading selects:

```tsx
const locationData = {
  USA: {
    California: ["Los Angeles", "San Francisco", "San Diego"],
    Texas: ["Houston", "Dallas", "Austin"]
  },
  Canada: {
    Ontario: ["Toronto", "Ottawa"],
    Quebec: ["Montreal", "Quebec City"]
  }
};

const MyForm = useForm({
  country: "",
  state: "",
  city: ""
});

function LocationForm() {
  const country = MyForm.useWatch("country");
  const state = MyForm.useWatch("state");

  const states = country ? Object.keys(locationData[country] || {}) : [];
  const cities = country && state ? locationData[country]?.[state] || [] : [];

  // Reset dependent fields when parent changes
  React.useEffect(() => {
    MyForm.formApi.setFieldsValue({ state: "", city: "" });
  }, [country]);

  React.useEffect(() => {
    MyForm.formApi.setFieldValue("city", "");
  }, [state]);

  return (
    <MyForm onSubmit={(state) => console.log(state)}>
      <MyForm.Item name="country">
        {({ value, onChange }) => (
          <select value={value} onChange={(e) => onChange(e.target.value)}>
            <option value="">Select Country</option>
            {Object.keys(locationData).map((c) => (
              <option key={c} value={c}>{c}</option>
            ))}
          </select>
        )}
      </MyForm.Item>

      <MyForm.Item name="state">
        {({ value, onChange }) => (
          <select
            value={value}
            onChange={(e) => onChange(e.target.value)}
            disabled={!country}
          >
            <option value="">Select State</option>
            {states.map((s) => (
              <option key={s} value={s}>{s}</option>
            ))}
          </select>
        )}
      </MyForm.Item>

      <MyForm.Item name="city">
        {({ value, onChange }) => (
          <select
            value={value}
            onChange={(e) => onChange(e.target.value)}
            disabled={!state}
          >
            <option value="">Select City</option>
            {cities.map((c) => (
              <option key={c} value={c}>{c}</option>
            ))}
          </select>
        )}
      </MyForm.Item>

      <button type="submit">Submit</button>
    </MyForm>
  );
}
```


## Form State Persistence

Save and restore form state.

### LocalStorage Persistence

```tsx
function useFormPersistence(form: any, storageKey: string) {
  // Load from localStorage on mount
  React.useEffect(() => {
    const saved = localStorage.getItem(storageKey);
    if (saved) {
      try {
        const state = JSON.parse(saved);
        form.formApi.setFieldsValue(state);
      } catch (e) {
        console.error("Failed to load form state:", e);
      }
    }
  }, []);

  // Save to localStorage on every change
  React.useEffect(() => {
    const unsubscribe = form.formApi.onFieldChange(() => {
      const state = form.formApi.getState();
      localStorage.setItem(storageKey, JSON.stringify(state));
    });

    return unsubscribe;
  }, []);

  // Clear storage
  const clearStorage = () => {
    localStorage.removeItem(storageKey);
    form.formApi.resetFields();
  };

  return { clearStorage };
}

// Usage
function MyComponent() {
  const MyForm = useForm({
    title: "",
    content: "",
    tags: [] as string[]
  });

  const { clearStorage } = useFormPersistence(MyForm, "draft-post");

  return (
    <div>
      <MyForm onSubmit={(state) => {
        console.log("Publishing:", state);
        clearStorage(); // Clear draft after publish
      }}>
        {/* form fields */}
      </MyForm>

      <button onClick={clearStorage}>Clear Draft</button>
    </div>
  );
}
```

### Auto-Save Draft

```tsx
function useAutoSave(form: any, saveFn: (state: any) => Promise<void>, delay = 2000) {
  const timeoutRef = React.useRef<NodeJS.Timeout>();
  const [isSaving, setIsSaving] = React.useState(false);
  const [lastSaved, setLastSaved] = React.useState<Date | null>(null);

  React.useEffect(() => {
    const unsubscribe = form.formApi.onFieldChange(() => {
      // Clear existing timeout
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }

      // Set new timeout for auto-save
      timeoutRef.current = setTimeout(async () => {
        setIsSaving(true);
        try {
          const state = form.formApi.getState();
          await saveFn(state);
          setLastSaved(new Date());
        } catch (error) {
          console.error("Auto-save failed:", error);
        } finally {
          setIsSaving(false);
        }
      }, delay);
    });

    return () => {
      unsubscribe();
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, []);

  return { isSaving, lastSaved };
}

// Usage
function BlogPostEditor() {
  const MyForm = useForm({
    title: "",
    content: ""
  });

  const saveDraft = async (state: any) => {
    await fetch("/api/drafts", {
      method: "POST",
      body: JSON.stringify(state)
    });
  };

  const { isSaving, lastSaved } = useAutoSave(MyForm, saveDraft);

  return (
    <div>
      <div className="status">
        {isSaving && <span>Saving...</span>}
        {lastSaved && <span>Last saved: {lastSaved.toLocaleTimeString()}</span>}
      </div>

      <MyForm onSubmit={(state) => console.log("Publishing:", state)}>
        {/* form fields */}
      </MyForm>
    </div>
  );
}
```

## FormApi Outside React

Access and manipulate form state outside React components.

### Global Form Instance

```tsx
// forms/AppSettings.ts
import { createForm } from 'react-any-shape-form';

export const AppSettingsForm = createForm({
  theme: "light",
  language: "en",
  notifications: true
});

// Can be used anywhere in your app
export function saveSettings() {
  const settings = AppSettingsForm.formApi.getState();
  localStorage.setItem("settings", JSON.stringify(settings));
}

export function loadSettings() {
  const saved = localStorage.getItem("settings");
  if (saved) {
    const settings = JSON.parse(saved);
    AppSettingsForm.formApi.setFieldsValue(settings);
  }
}

// In React components
function SettingsPanel() {
  const theme = AppSettingsForm.useWatch("theme");

  return (
    <AppSettingsForm>
      <AppSettingsForm.Item name="theme">
        {({ value, onChange }) => (
          <select value={value} onChange={(e) => onChange(e.target.value)}>
            <option value="light">Light</option>
            <option value="dark">Dark</option>
          </select>
        )}
      </AppSettingsForm.Item>
    </AppSettingsForm>
  );
}

// Outside React
document.getElementById("save-btn")?.addEventListener("click", () => {
  saveSettings();
});
```

### Event-Based Integration

```tsx
// Subscribe to form changes from anywhere
const unsubscribe = MyForm.formApi.onFieldChange((field, value) => {
  console.log(\`Field \${field} changed to:\`, value);

  // Trigger analytics
  analytics.track("form_field_changed", { field, value });
});

// Subscribe to validation errors
const unsubscribeErrors = MyForm.formApi.onError((errors) => {
  console.log("Validation errors:", errors);

  // Send to error tracking
  errorTracker.log("validation_errors", errors);
});

// Subscribe to specific field
const unsubscribeEmail = MyForm.formApi.onFieldChange((field, value) => {
  if (field === "email") {
    // Do something when email changes
    checkEmailAvailability(value);
  }
});

// Cleanup
unsubscribe();
unsubscribeErrors();
unsubscribeEmail();
```

## Best Practices

### 1. Separate Business Logic

Keep form logic separate from UI components:

```tsx
// hooks/useRegistrationForm.ts
export function useRegistrationForm() {
  const form = useForm({
    email: "",
    password: "",
    confirmPassword: ""
  });

  const handleSubmit = async (state: any) => {
    try {
      await registerUser(state);
      router.push("/dashboard");
    } catch (error) {
      form.formApi.setFieldError("email", [
        { errorText: "Registration failed", id: "1" }
      ]);
    }
  };

  return { form, handleSubmit };
}

// components/RegistrationForm.tsx
function RegistrationForm() {
  const { form, handleSubmit } = useRegistrationForm();

  return (
    <form.Form onSubmit={handleSubmit}>
      {/* UI only */}
    </form.Form>
  );
}
```

### 2. Use TypeScript

Define explicit types for better type safety:

```tsx
interface UserProfile {
  personalInfo: {
    firstName: string;
    lastName: string;
  };
  contact: {
    email: string;
    phone: string;
  };
  preferences: {
    newsletter: boolean;
    notifications: boolean;
  };
}

const MyForm = useForm<UserProfile>({
  personalInfo: {
    firstName: "",
    lastName: ""
  },
  contact: {
    email: "",
    phone: ""
  },
  preferences: {
    newsletter: false,
    notifications: true
  }
});

// TypeScript ensures correct types
const [personalInfo, setPersonalInfo] = MyForm.useField("personalInfo");
// personalInfo is typed as { firstName: string; lastName: string }
```

### 3. Optimize Re-renders

Use hooks strategically to minimize re-renders:

```tsx
//  re-renders when email changes
function MyComponent() {
  const email = MyForm.useWatch("email");
  return <div>{email}</div>;
}
```
